枚举所有子串, 对每个子串都进行回文判断
复杂度$O(N\cdot 2^N)$
```go
func partition(s string) (res [][]string) {
	n := len(s)
	var track []string
	var backtrack func(i int)
	backtrack = func(i int) {
		if i == n {
			res = append(res, slices.Clone(track))
			return
		}
	loop:
		for j := i; j < n; j++ {
			//cur := s[i:j]
			//fmt.Println(i, j, cur)
			for l, r := i, j; l <= r; l, r = l+1, r-1 {
				if s[l] != s[r] {
					continue loop
				}
			}
			track = append(track, s[i:j+1])
			backtrack(j+1)
			track = track[:len(track)-1]
		}
	}
	backtrack(0)
	return
}
```
