[718. 最长重复子数组 - 力扣（LeetCode）](https://leetcode.cn/problems/maximum-length-of-repeated-subarray/description/)
#hot100
相比于子序列, 子数组不能间隔
如果两个位置上值不一样了, 那么直接是0
```go
func findLength(nums1 []int, nums2 []int) int {
  m, n := len(nums1), len(nums2)
  dp := make([][]int, m+1)
  for i := range dp {
    dp[i] = make([]int, n+1)
  }
  for i := 1; i <= m; i++ {
    for j := 1; j <= n; j++ {
      if nums1[i-1] == nums2[j-1] {
        dp[i][j] = dp[i-1][j-1] + 1
      }
    }
  }
  res := 0
  for i := range m+1 {
    for j := range n+1 {
      res = max(res, dp[i][j])
    }
  }
  return res
}
```
